\newcommand{\deffunctline}[5]{{{#1}\colon {#2}\ni{#4}\mapsto{#5}\in{#3}}}
\newcommand{\deffunctdisp}[5]{%
\begin{aligned}%
      {#1}\,\colon {#2} &\xrightarrow{\hphantom{#1}}{#3}\\
       {#4} &\xmapsto{\hphantom{#1}}{#5}
\end{aligned}%
}
\newcommand{\deffunct}[5]{%
  \ifcase\mathstyle
% 0 : displaystyle (entre \[ \])
    \deffunctdisp{#1}{#2}{#3}{#4}{#5}
  \else
    \deffunctline{#1}{#2}{#3}{#4}{#5}
\fi}
